# Once
sudo apt install -y gnupg gnupg-agent scdaemon pcscd
ssh-keygen -K

# .bashrc
if [ "$SSH_AGENT_PID" = "" ]; then
        eval "$(ssh-agent -s)"
fi

function cleanup()
{
        eval "$(ssh-agent -k)"
}
trap cleanup EXIT

# Windows (gpg-agent.conf)
enable-ssh-support
enable-win32-openssh-support
use-standard-socket
default-cache-ttl 600
max-cache-ttl 7200
